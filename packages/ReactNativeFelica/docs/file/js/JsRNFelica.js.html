    import { NativeModules, DeviceEventEmitter } from 'react-native';
import { EventEmitter2 } from 'eventemitter2';
import FELICA_COMMAND from './JsRNFelica'
import Util from "./Util";

const { RNFelica } = NativeModules;

/**
 * Convert idm for argument to byte array
 * @param {(Array.<Number> | String)} idm
 * @returns {Array.<Number>}
 */
function argIdm2ByteIdm(idm){
  let byteIdm = null;
  if(Array.isArray(idm)){
    byteIdm = idm
  }else if (typeof idm === 'string' || idm instanceof String){
    byteIdm = Util.stringHex2Byte(idm)
  }else{
    return null
  }
  return byteIdm;
}
class JsRNFelica extends EventEmitter2 {
  /**
   *  Constructor for JsRNFelica
   */
  constructor() {
    super();
    DeviceEventEmitter.addListener(RNFelica.EVENT.FELICA_EVENT, (felicaInfo) => {
      this.emit('JS_EVENT_FELICA', felicaInfo);
    });
  }

  /**

   * @returns {*}
   */
  static haveNfc() {
    return RNFelica.haveNfc();
  }

  /**
   *
   * @returns {*}
   */
  static enableNfc() {
    return RNFelica.enableNfc();
  }

  /**
   *
   */
  static connect() {
    return RNFelica.connect();
  }

  /**
   *
   */
  static close() {
    return RNFelica.close();
  }

  /**
   *
   * @param data
   * @returns {*}
   */
  static transceive(data) {
    return RNFelica.transceive();
  }

  /**

    return RNFelica.transceiveString(data);
  }

  /**
   *
   * @param data
   * @returns {*}
   */
  static commonSendCmd(data){
    const sendData = [...data];
    sendData.unshift(0x00);
    sendData[0] = sendData.length;
    return JsRNFelica.transceive(sendData)
  }

  /**
   *
   * @param systemCode
   * @param requestCode
   * @param timeSlot
   * @returns {*}
   */
  static polling(systemCode, requestCode, timeSlot) {
    const sendData = [
      FELICA_COMMAND.CC_POLLING,
      (systemCode >>> 8) & 0xFF,
      systemCode & 0xFF,
      requestCode,
      timeSlot
    ];
    return JsRNFelica.commonSendCmd(sendData);
  }

  /**
   *
   * @param {(Array.<Number> | String)} idm
   * @param nodes
   * @returns {*}
   */
  static requestService(idm, nodes) {
    const byteIdm = argIdm2ByteIdm(idm);
    if(byteIdm === null){
      return Promise.reject(new Error("idm is invalid"));
    }
    const sendData = [
      FELICA_COMMAND.CC_REQUEST_SERVICE,
      ...byteIdm
    ];
    for(const node in nodes){
      sendData.push(node & 0xFF);
      sendData.push((node >>>  8) & 0xFF);
    }
    return JsRNFelica.commonSendCmd(sendData);
  }

  /**
   *
   * @param {(Array.<Number> | String)} idm
   * @returns {*}
   */
  static requestResponse(idm) {
    const byteIdm = argIdm2ByteIdm(idm);
    if(byteIdm === null){
      return Promise.reject(new Error("idm is invalid"));
    }
    const sendData = [
      FELICA_COMMAND.CC_REQUEST_RESPONCE,
      ...byteIdm
    ];
    return JsRNFelica.commonSendCmd(sendData);
  }

  /**
   *
   * @param {(Array.<Number> | String)} accessMode
   * @param serviceCodeIndex
   * @param blockNum
   * @param three_bit_length
   * @returns {number}
   */
  static makeBlockElement(accessMode, serviceCodeIndex, blockNum, three_bit_length = false){
    let data = ((accessMode << 4) | serviceCodeIndex) << 8;
    data |= (three_bit_length === true) ? 0x00 : 0x80;
    if(three_bit_length === true){
      data <<= 8
    }
    data |= blockNum;
    return data;
  }

  /**
   *
   * @param {(Array.<Number> | String)} idm
   * @param services
   * @param blockList
   * @returns {*}
   */
  static readWithoutEncryption(idm, services, blockList) {
    const byteIdm = argIdm2ByteIdm(idm);
    if(byteIdm === null){
      return Promise.reject(new Error("idm is invalid"));
    }
    const sendData = [
      FELICA_COMMAND.CC_READ_WITHOUT_ENCRYPTION,
      ...byteIdm,
    ];
    sendData.push(services.length);
    for(let i = 0; i < services.length; i++){
      sendData.push(services[i] & 0xFF);
      sendData.push((services[i] >>> 8) & 0xFF);
    }
    sendData.push(blockList.length);
    for(const block in blockList){
      sendData.push(block & 0xFF);
      sendData.push((block >>> 8) & 0xFF);
      if((block & 0x1000) === 0){
        sendData.push((block >>> 16) & 0xFF);
      }
    }
    return JsRNFelica.commonSendCmd(sendData);
  }

  /**
   *
   * @param {(Array.<Number> | String)} idm
   * @param {Array.<Number>} services
   * @param {Array.<Number>} blockList
   * @param {Array.<Number>} blockData
   * @returns {Promise}
   */
  static writeWithoutEncryption(idm, services, blockList, blockData) {
    const byteIdm = argIdm2ByteIdm(idm);
    if(byteIdm === null){
      return Promise.reject(new Error("idm is invalid"));
    }
    const sendData = [
      FELICA_COMMAND.CC_READ_WITHOUT_ENCRYPTION,
      ...byteIdm
    ];
    sendData.push(services.length);
    for(let i = 0; i < services.length; i++){
      sendData.push(services[i] & 0xFF);
      sendData.push((services[i] >>> 8) & 0xFF);
    }
    sendData.push(blockList.length);
    for(const block in blockList){
      sendData.push(block & 0xFF);
      sendData.push((block >>> 8) & 0xFF);
      if((block & 0x1000) === 0){
        sendData.push((block >>> 16) & 0xFF);
      }
    }
    return JsRNFelica.commonSendCmd(sendData);
  }
  /**
   *
   * @param {(Array.<Number> | String)} idm
   * @returns {*}
   */
  static searchService(idm) {
    const byteIdm = argIdm2ByteIdm(idm);
    if(byteIdm === null){
      return Promise.reject(new Error("idm is invalid"));
    }
    const sendData = [
      FELICA_COMMAND.CC_SEARCH_SERVICE_CODE,
      ...byteIdm
    ];
    return JsRNFelica.commonSendCmd(sendData);
  }

  /**
   *
   * @param {(Array.<Number> | String)} idm
   * @param idx
   * @returns {*}
   */
  static requestSystem(idm, idx) {
    const byteIdm = argIdm2ByteIdm(idm);
    if(byteIdm === null){
      return Promise.reject(new Error("idm is invalid"));
    }
    const sendData = [
      FELICA_COMMAND.CC_REQUEST_SYSTEM_CODE,
      ...byteIdm,
      idx & 0xFF,
      (idx >>> 8) & 0xFF
    ];
    return JsRNFelica.commonSendCmd(sendData);
  }
}
